Device discoveryÔÉÅ

This example is based on the tutorial found in [IntelCorporation20]. This example illustrates how to use SYCL backend API to identify and query information regarding the devices available for computing. Full source code of the example is shown below:

 1#include <iostream>
 2
 3//! [include_sycl_begin]
 4#include <sycl/sycl.hpp>
 5//! [include_sycl_end]
 6
 7//! [namespace_sycl_begin]
 8using namespace sycl;
 9//! [namespace_sycl_end]
10
11int main(void) {
12  //! [iterate_platforms_begin]
13  for (auto &platform : platform::get_platforms()) {
14    //! [iterate_platforms_end]
15    //! [print_platform_info_begin]
16    std::cout << "Platform: name = "
17              << platform.get_info<info::platform::name>()
18              << ", vendor = " << platform.get_info<info::platform::vendor>()
19              << ", backend = "
20              << detail::get_backend_name_no_vendor(platform.get_backend())
21              << ", version = " << platform.get_info<info::platform::version>()
22              << std::endl;
23    //! [print_platform_info_end]
24
25    //! [iterate_devices_begin]
26    for (auto &device : platform.get_devices())
27      //! [iterate_devices_end]
28      //! [print_device_info_begin]
29      std::cout
30          << "\tDevice: name = " << device.get_info<info::device::name>()
31          << ", vendor = " << device.get_info<info::device::vendor>()
32          << ", max_compute_units = "
33          << device.get_info<info::device::max_compute_units>()
34          << ", max_work_item_dimensions = "
35          << device.get_info<info::device::max_work_item_dimensions>() << " <"
36          << device.get_info<info::device::max_work_item_sizes<3>>().get(0)
37          << ", "
38          << device.get_info<info::device::max_work_item_sizes<3>>().get(1)
39          << ", "
40          << device.get_info<info::device::max_work_item_sizes<3>>().get(2)
41          << ">" << std::endl;
42    //! [print_device_info_end]
43  }
44
45  return 0;
46}

We start by including the sycl header file which provides definition of the SYCL backend API.

#include <sycl/sycl.hpp>

We then add the following line at the beginning of the source file to use the sycl:: namespace implicitly and reduce the amount of typing we have to do by avoiding the sycl:: namespace prefix:

using namespace sycl;

platform::get_platforms() function returns a std::vector containing all the available SYCL platforms in the system (host). We first iterate through all the entries in that vector using the following line:

for (auto &platform : platform::get_platforms()) {

We then query and print name, vendor, backend and version for each platform:

std::cout << "Platform: name = "
          << platform.get_info<info::platform::name>()
          << ", vendor = " << platform.get_info<info::platform::vendor>()
          << ", backend = "
          << detail::get_backend_name_no_vendor(platform.get_backend())
          << ", version = " << platform.get_info<info::platform::version>()
          << std::endl;

Then we iterate through all the devices in the current platform using the std::vector returned from platform.get_devices() function:

for (auto &device : platform.get_devices())

Similar to the platforms, we then print name, vendor, max compute units, max work dimensions and maximum work items supported in each dimension.

std::cout
    << "\tDevice: name = " << device.get_info<info::device::name>()
    << ", vendor = " << device.get_info<info::device::vendor>()
    << ", max_compute_units = "
    << device.get_info<info::device::max_compute_units>()
    << ", max_work_item_dimensions = "
    << device.get_info<info::device::max_work_item_dimensions>() << " <"
    << device.get_info<info::device::max_work_item_sizes<3>>().get(0)
    << ", "
    << device.get_info<info::device::max_work_item_sizes<3>>().get(1)
    << ", "
    << device.get_info<info::device::max_work_item_sizes<3>>().get(2)
    << ">" << std::endl;
